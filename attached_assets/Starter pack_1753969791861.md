## User Flow Implementation

### Selection and Creation Process

```typescript
const handleSelectStarterPack = (packId: string) => {
  const allPacks = Object.values(starterPacksByCategory).flatMap(category => category.packs);
  const pack = allPacks.find(p => p.id === packId);
  if (!pack) return;

  setSelectedPack(packId);
  setIsLoading(true);

  setTimeout(() => {
    // Route based on integration context
    if (onSelectTemplate) {
      // Returning user flow
      onSelectTemplate({
        title: pack.title,
        description: pack.description,
        color: pack.color,
        members: pack.members,
        welcomeMessage: pack.welcomeMessage
      });
    } else if (onDirectComplete) {
      // Onboarding flow
      onDirectComplete(pack.title, templateData);
    } else if (onAddTeam && activeProject) {
      // Direct team creation
      const { team, agents } = createCustomTeamFromPack(pack);
      if (team) {
        onAddTeam(team, agents);
      }
    }

    if (showAnimation) {
      showAnimation();
    }

    setIsLoading(false);
    if (onComplete) {
      onComplete(pack.title);
    }
  }, 500);
};
```

### Team Creation from Template

```typescript
const createCustomTeamFromPack = (pack: any, isExploration = false) => {
  if (!activeProject) return { team: null, agents: [] };
  
  const teamId = uuidv4();
  const colors = ["blue", "green", "purple", "amber"];
  const randomColor = colors[Math.floor(Math.random() * colors.length)];
  
  const newTeam: Team = {
    id: teamId,
    name: pack.title,
    description: pack.description,
    projectId: activeProject.id,
    color: pack.color || randomColor,
    dateCreated: new Date()
  };

  const newAgents: Agent[] = pack.members.map((memberName: string) => {
    const hatchTemplate = getHatchDetails(memberName);
    return {
      id: uuidv4(),
      name: memberName,
      role: hatchTemplate?.role || "Team Member",
      description: hatchTemplate?.description || `Member of the ${pack.title} team`,
      color: hatchTemplate?.color || pack.color || randomColor,
      projectId: activeProject.id,
      teamId: teamId
    };
  });

  return { team: newTeam, agents: newAgents };
};
```
